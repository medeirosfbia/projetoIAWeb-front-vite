import { ChangeEvent, useEffect, useState } from "react";
import { useNavigate } from "react-router-dom";
import User from "../../models/User";
import { createUser } from "../../services/AuthService";

function Register() {

    const navigate = useNavigate();

    const [confirmPassword, setConfirmPassword] = useState<string>('');

    const [usuario, setUser] = useState<User>({
        id: 0,
        usuario: '',
        email: '',
        nome: '',
        sobrenome: '',
        senha: '',
        nascimento: '',
        foto: '',
    });

    const[userBackResp, setUserBackResp] = useState<User>({
        id: 0,
        usuario: '',
        email: '',
        nome: '',
        sobrenome: '',
        senha: '',
        nascimento: '',
        foto: '',
    });

    function updateState(e: ChangeEvent<HTMLInputElement>) {
        setUser({...usuario,[e.target.name]: e.target.value});
    }

    function handleConfirmPassword(e: ChangeEvent<HTMLInputElement>) {
        setConfirmPassword(e.target.value);
    }

    async function submitNewUser(event: ChangeEvent<HTMLFormElement>) {
        event.preventDefault();
        if(confirmPassword === usuario.senha ){
            try {
                await createUser('/user/signup', usuario, setUserBackResp)
                alert("Usuário cadastrado com sucesso!")
            } catch (error) {
                alert("Erro ao cadastrar usuário!")
            }
        } else{
            alert("Senhas não conferem!")
            setUser({...usuario, senha:''})
            setConfirmPassword('')
        }
    }

function back(){
    navigate("/login")
}
